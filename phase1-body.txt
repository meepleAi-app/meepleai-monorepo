## 🔧 Technical Details

### useLocalStorage Implementation

```typescript
// apps/web/src/hooks/useLocalStorage.ts
export function useLocalStorage<T>(
  key: string,
  defaultValue: T
): [T, (value: T | ((prev: T) => T)) => void] {
  const [storedValue, setStoredValue] = useState<T>(() => {
    // SSR-safe: return default during SSR
    if (typeof window === 'undefined') {
      return defaultValue;
    }

    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : defaultValue;
    } catch (error) {
      console.warn(`Error reading localStorage key "${key}":`, error);
      return defaultValue;
    }
  });

  const setValue = useCallback((value: T | ((prev: T) => T)) => {
    try {
      const valueToStore = value instanceof Function ? value(storedValue) : value;
      setStoredValue(valueToStore);

      if (typeof window !== 'undefined') {
        window.localStorage.setItem(key, JSON.stringify(valueToStore));
        // Dispatch storage event for cross-tab sync
        window.dispatchEvent(new StorageEvent('storage', {
          key,
          newValue: JSON.stringify(valueToStore),
        }));
      }
    } catch (error) {
      console.error(`Error setting localStorage key "${key}":`, error);
    }
  }, [key, storedValue]);

  // Cross-tab synchronization
  useEffect(() => {
    const handleStorageChange = (e: StorageEvent) => {
      if (e.key === key && e.newValue) {
        try {
          setStoredValue(JSON.parse(e.newValue));
        } catch (error) {
          console.warn(`Error parsing storage event for key "${key}":`, error);
        }
      }
    };

    window.addEventListener('storage', handleStorageChange);
    return () => window.removeEventListener('storage', handleStorageChange);
  }, [key]);

  return [storedValue, setValue];
}
```

### ViewModeSwitcher Component

```typescript
// apps/web/src/components/ui/data-display/entity-list-view/components/view-mode-switcher.tsx
import { Grid3x3, List, GalleryHorizontal } from 'lucide-react';

type ViewMode = 'grid' | 'list' | 'carousel';

interface ViewModeSwitcherProps {
  value: ViewMode;
  onChange: (mode: ViewMode) => void;
  availableModes?: ViewMode[];
  className?: string;
}

const MODE_CONFIG: Record<ViewMode, { icon: any; label: string }> = {
  grid: { icon: Grid3x3, label: 'Grid' },
  list: { icon: List, label: 'List' },
  carousel: { icon: GalleryHorizontal, label: 'Carousel' },
};

export function ViewModeSwitcher({
  value,
  onChange,
  availableModes = ['grid', 'list', 'carousel'],
  className,
}: ViewModeSwitcherProps) {
  const handleKeyDown = (e: React.KeyboardEvent, modes: ViewMode[]) => {
    const currentIdx = modes.indexOf(value);

    if (e.key === 'ArrowRight' || e.key === 'ArrowDown') {
      e.preventDefault();
      const nextIdx = (currentIdx + 1) % modes.length;
      onChange(modes[nextIdx]);
    } else if (e.key === 'ArrowLeft' || e.key === 'ArrowUp') {
      e.preventDefault();
      const prevIdx = (currentIdx - 1 + modes.length) % modes.length;
      onChange(modes[prevIdx]);
    }
  };

  return (
    <div
      role="radiogroup"
      aria-label="View mode selection"
      className={cn(
        'inline-flex items-center gap-1',
        'p-1 rounded-lg',
        'bg-muted/20 border border-border/50',
        className
      )}
      onKeyDown={(e) => handleKeyDown(e, availableModes)}
    >
      {availableModes.map((mode) => {
        const { icon: Icon, label } = MODE_CONFIG[mode];
        const isActive = value === mode;

        return (
          <button
            key={mode}
            onClick={() => onChange(mode)}
            role="radio"
            aria-checked={isActive}
            aria-label={`${label} view`}
            className={cn(
              'flex items-center gap-2 px-3 py-2 rounded-md',
              'transition-all duration-200',
              'focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring',
              isActive
                ? 'bg-primary/10 text-primary border-b-2 border-primary'
                : 'text-muted-foreground hover:bg-muted/40 hover:text-foreground'
            )}
          >
            <Icon className="w-4 h-4" />
            <span className="text-sm font-medium hidden sm:inline">{label}</span>
          </button>
        );
      })}
    </div>
  );
}
```

### EntityListView Scaffold (Grid Only)

```typescript
// apps/web/src/components/ui/data-display/entity-list-view/entity-list-view.tsx
export function EntityListView<T = any>({
  items,
  entity,
  persistenceKey,
  defaultViewMode = 'grid',
  renderItem,
  onItemClick,
  gridColumns = { default: 1, sm: 2, lg: 3, xl: 4 },
  gridGap = 4,
  title,
  subtitle,
  showViewSwitcher = true,
  emptyMessage = 'No items to display',
  loading = false,
  className,
  'data-testid': testId,
}: EntityListViewProps<T>) {
  const { mode, setMode } = useViewMode(persistenceKey, defaultViewMode, ['grid', 'list', 'carousel']);

  // Phase 1: Only grid mode supported
  const renderGridLayout = () => (
    <div
      className={cn(
        'grid gap-4',
        gridColumns.default && `grid-cols-${gridColumns.default}`,
        gridColumns.sm && `sm:grid-cols-${gridColumns.sm}`,
        gridColumns.lg && `lg:grid-cols-${gridColumns.lg}`,
        gridColumns.xl && `xl:grid-cols-${gridColumns.xl}`,
        `gap-${gridGap}`
      )}
      data-testid="grid-layout"
    >
      {items.map((item, idx) => {
        const cardProps = renderItem(item);
        return (
          <MeepleCard
            key={cardProps.id || idx}
            entity={entity}
            variant="grid"
            {...cardProps}
            onClick={() => onItemClick?.(item)}
          />
        );
      })}
    </div>
  );

  if (loading) {
    return <LoadingSkeleton mode={mode} />;
  }

  if (items.length === 0) {
    return <EmptyState message={emptyMessage} />;
  }

  return (
    <section className={cn('w-full', className)} data-testid={testId || 'entity-list-view'}>
      {/* Header */}
      {(title || subtitle || showViewSwitcher) && (
        <header className="flex items-start justify-between mb-6">
          <div>
            {title && <h2 className="font-quicksand font-bold text-2xl md:text-3xl">{title}</h2>}
            {subtitle && <p className="mt-2 text-muted-foreground text-sm md:text-base">{subtitle}</p>}
          </div>
          {showViewSwitcher && (
            <ViewModeSwitcher
              value={mode}
              onChange={setMode}
              availableModes={['grid']} // Phase 1: only grid
            />
          )}
        </header>
      )}

      {/* Content - Phase 1: Grid only */}
      {renderGridLayout()}
    </section>
  );
}
```
