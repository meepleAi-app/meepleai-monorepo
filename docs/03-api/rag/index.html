🎓 How TOMAC-RAG Works - Complete Guide

What is TOMAC-RAG? Token-Optimized Modular Adaptive Corrective RAG is MeepleAI's intelligent question-answering system for board game rules. It optimizes the three-way tradeoff between Cost (token consumption), Accuracy (answer quality), and Latency (response time).

The Core Problem

A naive RAG system treats all queries the same way:

Every query → Retrieve 10 chunks → Send to expensive LLM → Generate answer

This wastes tokens on simple questions and underperforms on complex ones.

TOMAC-RAG's Solution

Intelligently route each query to the optimal strategy based on:

  1. Who's asking (user tier determines budget)
  2. What they're asking (template type: rules vs strategy)
  3. How complex it is (complexity score 0-5)

How it works (T5-Large evaluator, NOT LLM):

  1. Evaluate each document: Score 0-1 relevance
  2. Classify: Correct (≥0.8) / Ambiguous (0.5-0.8) / Incorrect (<0.5)

class AgentPromptBuilder:
    def build_prompt(
        self,
        agent_role: AgentRole,
        template: str,           # "rule_lookup" | "resource_planning"
        rag_context: List[Chunk],  # From L3-L4
        user_query: str,
        strategy: Strategy       # FAST | BALANCED | PRECISE
    ) -> str:

        # 1. SYSTEM PROMPT (fixed per agent role)
        system = self.get_agent_system_prompt(agent_role)

        # 2. RAG CONTEXT (dynamic, from retrieval pipeline)
        context = self.format_chunks(rag_context)

        # 3. USER QUERY (preserved verbatim)
        query = user_query

        # 4. TEMPLATE INSTRUCTIONS (based on query type)
        instructions = self.get_template_instructions(template)

        # 5. OUTPUT FORMAT (based on strategy)
        output_format = self.get_output_format(strategy)

        return f"""
{system}

## Contesto Rilevante
{context}

## Domanda Utente
{query}

## Istruzioni
{instructions}

## Formato Output
{output_format}
"""
Key Insight: The prompt is NOT created independently of RAG. Instead, RAG results (L3-L4) are injected into the prompt at L5. The routing decision (L1) determines which prompt template and output format to use. This is how agent behavior adapts to query complexity.

Mapping Strategia → Modello (Logica di Selezione)

public class RagModelSelector
{
    /// 
    /// Seleziona il modello ottimale basandosi su strategia, tier e budget.
    /// 
    public async Task<string> SelectModelAsync(
        RagStrategy strategy,
        LlmUserTier userTier,
        CancellationToken ct)
    {
        // 1. Verifica budget disponibile
        var budgetStatus = await _budgetService.GetStatusAsync(ct);
        if (budgetStatus.IsOverBudget)
        {
            // Auto-downgrade a modello gratuito
            return "meta-llama/llama-3.3-70b-instruct:free";
        }

        // 2. Ottieni configurazione tier dal database
        var tierConfig = await _tierRoutingService.GetRoutingAsync(userTier, ct);
        if (tierConfig?.ProductionModelId != null)
        {
            return tierConfig.ProductionModelId;
        }

        // 3. Fallback: Selezione basata su strategia
        return strategy switch

Layer 1: Routing Implementation

from dataclasses import dataclass
from enum import Enum

class Strategy(Enum):
    FAST = "fast"
    BALANCED = "balanced"
    PRECISE = "precise"
    EXPERT = "expert"      # Web search + multi-hop
    CONSENSUS = "consensus"  # Multi-LLM voting
    CUSTOM = "custom"      # Admin-defined

class Template(Enum):
    RULE_LOOKUP = "rule_lookup"
    RESOURCE_PLANNING = "resource_planning"

@dataclass
class RoutingDecision:
    strategy: Strategy
    template: Template
    complexity: int  # 0-5
    model: str
    user_tier: str
    tokens_budget: int

async def route_query(query: str, user: User) -> RoutingDecision:
    """3-dimensional routing: User tier + Template + Complexity"""

    # Dimension 1: User tier determines budget
    tier_budgets = {
        "Anonymous": 1500, "User": 3000, "Editor": 5000, "Admin": 15000
    }
    budget = tier_budgets.get(user.role, 1500)

    # Dimension 2: Template classification (Semantic Router or LLM)
    template = await classify_template(query)

    # Dimension 3: Complexity scoring
    complexity = calc_complexity(query, template)

    # Strategy selection
    if complexity <= 1 and template == Template.RULE_LOOKUP:
        strategy = Strategy.FAST
    elif complexity <= 3:
        strategy = Strategy.BALANCED
    else:
        strategy = Strategy.PRECISE

    # Model selection based on tier + strategy
    model = select_model(user.role, strategy)

    return RoutingDecision(
        strategy=strategy,
        template=template,
        complexity=complexity,
        model=model,
        user_tier=user.role,
        tokens_budget=budget
    )

def calc_complexity(query: str, template: Template) -> int:
    """Score complexity 0-5 based on query characteristics"""
    score = 0
    words = len(query.split())

    # Length-based scoring
    if words > 50: score += 2
    elif words > 25: score += 1

    # Multi-concept keywords
    if any(k in query.lower() for k in ["and", "or", "both"]): score += 1

    # Edge cases / conditionals
    if any(k in query.lower() for k in ["what if", "except when", "unless"]): score += 1

    # Negation complexity
    if any(k in query.lower() for k in ["not", "don't", "can't", "never"]): score += 1

    # Template baseline (resource_planning is inherently more complex)
    if template == Template.RESOURCE_PLANNING: score += 1

    return min(score, 5)

Layer 2: Semantic Cache Implementation

import redis.asyncio as redis
import anthropic
from dataclasses import dataclass
import hashlib
import json

@dataclass
class CacheResult:
    hit: bool
    answer: str | None = None
    tokens_used: int = 0

class SemanticCache:
    def __init__(self, redis_url: str, anthropic_api_key: str):
        self.redis = redis.from_url(redis_url)
        self.client = anthropic.AsyncAnthropic(api_key=anthropic_api_key)

    async def lookup(self, query: str) -> CacheResult:
        """Check if similar query exists in cache using LLM similarity"""

        # Get recent cached queries
        recent = await self.redis.lrange("rag:recent", 0, 10)
        if not recent:
            return CacheResult(hit=False)

        # LLM-based semantic similarity check (310 tokens)
        prompt = f"""
        Is this new query semantically similar to any cached query?
        New: {query}
        Cached: {[r.decode() for r in recent]}

        Return index (0-{len(recent)-1}) of most similar, or null if none.
        Similarity threshold: 0.85
        """

        response = await self.client.messages.create(
            model="claude-3-haiku-20240307",
            max_tokens=10,
            messages=[{"role": "user", "content": prompt}]
        )

        try:
            idx = int(response.content[0].text.strip())
            cached_query = recent[idx].decode()
            cache_key = f"rag:{hashlib.md5(cached_query.encode()).hexdigest()}"
            cached_data = await self.redis.get(cache_key)

            if cached_data:
                data = json.loads(cached_data)
                return CacheResult(hit=True, answer=data["answer"], tokens_used=310)
        except:
            pass

        return CacheResult(hit=False, tokens_used=310)

    async def store(self, query: str, answer: str, ttl_hours: int = 24):
        """Cache query-answer pair with TTL"""
        cache_key = f"rag:{hashlib.md5(query.encode()).hexdigest()}"
        data = json.dumps({"query": query, "answer": answer})
        await self.redis.setex(cache_key, ttl_hours * 3600, data)
        await self.redis.lpush("rag:recent", query)
        await self.redis.ltrim("rag:recent", 0, 99)  # Keep last 100

        self.collection = collection
        self.embed_fast = SentenceTransformer("all-MiniLM-L6-v2")      # 14.7ms
        self.embed_balanced = SentenceTransformer("intfloat/e5-base-v2")  # 79ms
        self.embed_precise = SentenceTransformer("BAAI/bge-base-en-v1.5")  # 50ms

    async def retrieve(self, query: str, strategy: str, game_id: str = None) -> List[Document]:
        """Route to appropriate retrieval strategy"""
        if strategy == "FAST":
            return await self._retrieve_fast(query, game_id)
        elif strategy == "BALANCED":
            return await self._retrieve_balanced(query, game_id)
        else:  # PRECISE
            return await self._retrieve_precise(query, game_id)

    async def _retrieve_fast(self, query: str, game_id: str = None) -> List[Document]:
        """FAST: Vector-only retrieval, Top-3 (~1,500 tokens)"""
        embedding = self.embed_fast.encode(query).tolist()

        filter_condition = None
        if game_id:
            filter_condition = models.Filter(
                must=[models.FieldCondition(key="game_id", match=models.MatchValue(value=game_id))]
            )

        results = self.qdrant.search(
            collection_name=self.collection,
            query_vector=embedding,
            limit=3,  # Top-3 only for FAST
            query_filter=filter_condition
        )

        return [Document(
            id=str(r.id),
            content=r.payload["text"],
            metadata=r.payload,
            score=r.score
        ) for r in results]

    async def _retrieve_balanced(self, query: str, game_id: str = None) -> List[Document]:
        """BALANCED: Hybrid search (Vector + BM25) with RRF (~3,500 tokens)"""
        embedding = self.embed_balanced.encode(query).tolist()

        # Parallel: Vector + Keyword search
        vector_task = asyncio.create_task(self._vector_search(embedding, game_id, limit=10))
        keyword_task = asyncio.create_task(self._keyword_search(query, game_id, limit=10))

        vector_results, keyword_results = await asyncio.gather(vector_task, keyword_task)

        # Reciprocal Rank Fusion (RRF)
        fused = self._reciprocal_rank_fusion(vector_results, keyword_results, k=60)

        return fused[:10]  # Top-10 after fusion

    async def _retrieve_precise(self, query: str, game_id: str = None, max_hops: int = 3) -> List[Document]:
        """PRECISE: Multi-hop adaptive retrieval (~8,000 tokens)"""
        all_docs = []
        seen_ids = set()

        # Hop 1: Initial broad retrieval
        embedding = self.embed_precise.encode(query).tolist()
        hop1 = await self._vector_search(embedding, game_id, limit=20)
        all_docs.extend(hop1)
        seen_ids.update(d.id for d in hop1)

        # Hop 2: Entity expansion
        entities = self._extract_entities(hop1)
        for entity in entities[:5]:
            entity_embedding = self.embed_precise.encode(entity).tolist()
            hop2_results = await self._vector_search(entity_embedding, game_id, limit=5, exclude_ids=seen_ids)
            all_docs.extend(hop2_results)
            seen_ids.update(d.id for d in hop2_results)

        # Hop 3: Cross-reference validation
        related = await self._retrieve_related_rules(all_docs, seen_ids)
        all_docs.extend(related)

        # Deduplicate and return top-20
        return self._deduplicate(all_docs)[:20]

    def _reciprocal_rank_fusion(self, list1: List[Document], list2: List[Document], k: int = 60) -> List[Document]:
        """Combine rankings using RRF: score = sum(1 / (k + rank))"""
        scores = {}
        doc_map = {}

        for rank, doc in enumerate(list1, 1):
            scores[doc.id] = scores.get(doc.id, 0) + 1 / (k + rank)
            doc_map[doc.id] = doc

        for rank, doc in enumerate(list2, 1):
            scores[doc.id] = scores.get(doc.id, 0) + 1 / (k + rank)
            doc_map[doc.id] = doc

        sorted_ids = sorted(scores.keys(), key=lambda x: scores[x], reverse=True)
        return [doc_map[id] for id in sorted_ids]

    def _extract_entities(self, docs: List[Document]) -> List[str]:
        """Extract game mechanics and terms from documents"""
        entities = set()
        for doc in docs:
            # Simple extraction - production would use NER
            if "mechanics" in doc.metadata:
                entities.update(doc.metadata["mechanics"])
            if "phase" in doc.metadata:
                entities.add(doc.metadata["phase"])
        return list(entities)

    def _deduplicate(self, docs: List[Document]) -> List[Document]:
        """Remove duplicate documents by ID"""
        seen = set()
        unique = []
        for doc in docs:
            if doc.id not in seen:
                seen.add(doc.id)
                unique.append(doc)
        return unique

from transformers import T5ForSequenceClassification, T5Tokenizer
from dataclasses import dataclass
from enum import Enum

class DocCategory(Enum):
    CORRECT = "correct"      # score >= 0.8
    AMBIGUOUS = "ambiguous"  # 0.5 <= score < 0.8
    INCORRECT = "incorrect"  # score < 0.5

@dataclass
class CragResult:
    docs: list
    source: str  # "internal" | "internal+web" | "web"
    tokens_saved: int

class CragEvaluator:
    def __init__(self, model_path: str):
        self.model = T5ForSequenceClassification.from_pretrained(model_path)
        self.tokenizer = T5Tokenizer.from_pretrained(model_path)

    async def evaluate(self, query: str, docs: list) -> CragResult:
        """Evaluate document relevance and route accordingly"""

        evaluations = []
        for doc in docs:
            score = self._score_relevance(query, doc.content)

            if score >= 0.8:
                category = DocCategory.CORRECT
            elif score >= 0.5:
                category = DocCategory.AMBIGUOUS
            else:
                category = DocCategory.INCORRECT

            evaluations.append({"doc": doc, "category": category, "score": score})

        # Count by category
        correct = [e for e in evaluations if e["category"] == DocCategory.CORRECT]
        ambiguous = [e for e in evaluations if e["category"] == DocCategory.AMBIGUOUS]

        # Decision logic
        if len(correct) >= 3:
            # Sufficient good docs - use internal only
            final_docs = [e["doc"] for e in sorted(correct, key=lambda x: -x["score"])[:5]]
            source = "internal"
        elif correct or ambiguous:
            # Partial info - augment with web search
            web_docs = await self._web_search(query + " official rules FAQ")
            combined = correct + ambiguous + [{"doc": d, "score": 0.7} for d in web_docs]
            final_docs = await self._rerank(combined, query, top_k=5)
            source = "internal+web"
        else:
            # All docs irrelevant - web search only
            web_docs = await self._web_search(query + " board game rules")
            final_docs = await self._rerank(web_docs, query, top_k=10)
            source = "web"

        # Decompose-recompose: extract only key sentences (40-70% reduction)
        filtered_docs = self._decompose_recompose(final_docs, query)
        tokens_saved = sum(len(d.content.split()) for d in docs) - \
                       sum(len(d.content.split()) for d in filtered_docs)

        return CragResult(docs=filtered_docs, source=source, tokens_saved=tokens_saved)

    def _decompose_recompose(self, docs: list, query: str) -> list:
        """Extract only top 60% most relevant sentences from each doc"""
        filtered = []
        for doc in docs:
            sentences = doc.content.split('. ')
            ranked = self._rank_sentences(sentences, query)
            keep_count = int(len(sentences) * 0.6)
            top_sentences = ranked[:keep_count]
            filtered.append(Document(content='. '.join(top_sentences)))
        return filtered

Layer 6: Validation & Auto-Escalation

import re
import json
from dataclasses import dataclass
from typing import Optional, List
from enum import Enum
from sentence_transformers import CrossEncoder

class ValidationAction(Enum):
    PASS = "pass"
    ESCALATE_TO_BALANCED = "escalate_balanced"
    ESCALATE_TO_PRECISE = "escalate_precise"
    OFFER_ESCALATE = "offer_escalate"

@dataclass
class ValidationResult:
    passed: bool
    confidence: float
    action: ValidationAction = ValidationAction.PASS
    reflection: Optional[dict] = None
    reason: Optional[str] = None

class StrategyValidator:
    """
    Validation by strategy:
    - FAST: Rule-based (0 LLM tokens)
    - BALANCED: Cross-encoder alignment (0 LLM tokens)
    - PRECISE: Self-RAG reflection (~4,400 tokens)
    """

    def __init__(self, cross_encoder_model: str, llm_client):
        self.cross_encoder = CrossEncoder(cross_encoder_model)
        self.llm = llm_client

    async def validate(self, answer: str, docs: List, query: str,
                       template: str, strategy: str) -> ValidationResult:
        """Route to appropriate validation strategy"""
        if strategy == "FAST":
            return self._validate_fast(answer, template)
        elif strategy == "BALANCED":
            return await self._validate_balanced(answer, docs)
        else:  # PRECISE
            return await self._validate_precise(answer, docs, query)

    def _validate_fast(self, answer: str, template: str) -> ValidationResult:
        """
        FAST: Rule-based validation (0 LLM tokens)
        - Check for citations in rule_lookup
        - Check minimum answer length
        - Auto-escalate to BALANCED if fails (~10-15% queries)
        """
        # Citation check for rule_lookup template
        if template == "rule_lookup":
            has_citation = bool(re.search(r'\(p(?:age)?\.?\s*\d+\)', answer, re.IGNORECASE))
            if not has_citation:
                return ValidationResult(
                    passed=False,
                    confidence=0.5,
                    action=ValidationAction.ESCALATE_TO_BALANCED,
                    reason="Missing page citation for rule lookup"
                )

        # Minimum length check
        word_count = len(answer.split())
        if word_count < 10:
            return ValidationResult(
                passed=False,
                confidence=0.4,
                action=ValidationAction.ESCALATE_TO_BALANCED,
                reason=f"Answer too short ({word_count} words)"
            )

        # Forbidden phrase check
        forbidden = ["i don't know", "unclear", "not sure", "cannot determine"]
        if any(phrase in answer.lower() for phrase in forbidden):
            return ValidationResult(
                passed=False,
                confidence=0.3,
                action=ValidationAction.ESCALATE_TO_BALANCED,
                reason="Answer contains uncertainty phrases"
            )

        return ValidationResult(passed=True, confidence=0.75)

    async def _validate_balanced(self, answer: str, docs: List) -> ValidationResult:
        """
        BALANCED: Cross-encoder alignment (0 LLM tokens)
        - Score answer-document pairs for semantic alignment
        - Offer escalate to PRECISE if alignment < 0.6 (~5% queries)
        """
        if not docs:
            return ValidationResult(passed=True, confidence=0.5)

        # Score answer against each document
        pairs = [(answer, doc.content) for doc in docs]
        scores = self.cross_encoder.predict(pairs)

        avg_score = sum(scores) / len(scores)
        max_score = max(scores)

        # Low alignment - offer PRECISE escalation
        if avg_score < 0.6:
            return ValidationResult(
                passed=True,  # Still pass, but offer escalation
                confidence=avg_score,
                action=ValidationAction.OFFER_ESCALATE,
                reason=f"Low alignment ({avg_score:.2f}), consider PRECISE for better accuracy"
            )

        return ValidationResult(passed=True, confidence=max(avg_score, max_score))

    async def _validate_precise(self, answer: str, docs: List, query: str) -> ValidationResult:
        """
        PRECISE: Self-RAG reflection (~4,400 tokens)
        - LLM self-critique with reflection tokens
        - Evaluate: Relevance, Support, Usefulness
        - Re-retrieve if confidence < 0.9 (max 2 loops)
        """
        docs_text = "\n\n".join([f"[{i+1}] {d.content}" for i, d in enumerate(docs)])

        reflection_prompt = f"""Self-critique your answer using reflection tokens.

## Your Answer
{answer}

## Retrieved Context
{docs_text}

## Original Question
{query}

## Evaluation Criteria
Rate each criterion and provide JSON:

1. **Relevance**: Does the answer address the question?
   - [Relevant]: Directly answers the question
   - [Partially]: Addresses some aspects
   - [Not Relevant]: Doesn't address the question

2. **Support**: Is the answer grounded in the context?
   - [Supported]: All claims backed by context
   - [Partially]: Some claims backed
   - [Not Supported]: Claims not in context

3. **Usefulness**: Is the answer helpful for the user?
   - [Useful]: Actionable and clear
   - [Partially]: Somewhat helpful
   - [Not Useful]: Vague or unhelpful

4. **Confidence**: Overall confidence score (0.0-1.0)

5. **Re-retrieve**: Should we retrieve more documents? (true/false)

Return JSON:
{{"relevance": "[Relevant|Partially|Not Relevant]", "support": "[Supported|Partially|Not Supported]", "usefulness": "[Useful|Partially|Not Useful]", "confidence": float, "re_retrieve": bool, "issues": ["issue1", "issue2"]}}"""

        response = await self.llm.generate(reflection_prompt, max_tokens=300)
        try:
            reflection = json.loads(response.content)
        except json.JSONDecodeError:
            # Fallback if LLM doesn't return valid JSON
            reflection = {"confidence": 0.7, "re_retrieve": False, "issues": []}

        # Calculate weighted confidence
        relevance_score = {"[Relevant]": 1.0, "[Partially]": 0.5, "[Not Relevant]": 0.0}.get(
            reflection.get("relevance", ""), 0.5
        )
        support_score = {"[Supported]": 1.0, "[Partially]": 0.5, "[Not Supported]": 0.0}.get(
            reflection.get("support", ""), 0.5
        )
        usefulness_score = {"[Useful]": 1.0, "[Partially]": 0.5, "[Not Useful]": 0.0}.get(
            reflection.get("usefulness", ""), 0.5
        )

        final_confidence = (
            0.4 * relevance_score +
            0.35 * support_score +
            0.25 * usefulness_score
        ) * reflection.get("confidence", 0.7)

        return ValidationResult(
            passed=final_confidence >= 0.7,
            confidence=final_confidence,
            reflection=reflection,
            reason="; ".join(reflection.get("issues", []))
        )

    def verify_citations(self, answer: str, docs: List) -> bool:
        """Verify that cited pages exist in retrieved documents"""
        cited_pages = re.findall(r'page\s+(\d+)', answer, re.IGNORECASE)
        for page in cited_pages:
            page_num = int(page)
            if not any(page_num == d.metadata.get("page") for d in docs):
                return False
        return True

Complete End-to-End Pipeline

import asyncio
import time
from dataclasses import dataclass
from typing import Optional

@dataclass
class TomacRagResponse:
    answer: str
    citations: list
    confidence: float
    strategy_used: str
    cache_hit: bool
    tokens_consumed: int
    cost_usd: float
    latency_ms: int
    trace: dict  # Full execution trace for debugging

class TomacRagPipeline:
    """
    Complete TOMAC-RAG Pipeline orchestrating all 6 layers:
    L1: Routing → L2: Cache → L3: Retrieval → L4: CRAG → L5: Generation → L6: Validation
    """

    def __init__(self, config: dict):
        # Initialize all layer components
        self.router = StrategyRouter(config["routing"])           # Layer 1
        self.cache = SemanticCache(config["cache"])               # Layer 2
        self.retriever = ModularRetriever(config["retrieval"])    # Layer 3
        self.crag = CragEvaluator(config["crag"])                 # Layer 4
        self.generator = AgentPromptBuilder(config["generation"]) # Layer 5
        self.validator = StrategyValidator(config["validation"])  # Layer 6
        self.multi_agent = MultiAgentOrchestrator(config["agents"]) # PRECISE only
        self.llm_clients = config["llm_clients"]

    async def ask(self, query: str, user_id: str, game_id: str = None,
                  template: str = "rule_lookup") -> TomacRagResponse:
        """Execute full TOMAC-RAG pipeline with all 6 layers"""

        start_time = time.time()
        trace = {"layers": {}}  # Track execution for debugging

        # ==== LAYER 1: ROUTING ====
        user_tier = await self._get_user_tier(user_id)
        complexity = self._estimate_complexity(query)
        strategy = self.router.select_strategy(user_tier, template, complexity)

        trace["layers"]["L1_routing"] = {
            "user_tier": user_tier,
            "template": template,
            "complexity": complexity,
            "strategy": strategy,
            "tokens": 320
        }

        # ==== LAYER 2: CACHE ====
        cache_result = await self.cache.lookup(query)

        if cache_result.hit:
            trace["layers"]["L2_cache"] = {"hit": True, "tokens": 310}
            return TomacRagResponse(
                answer=cache_result.answer,
                citations=cache_result.citations,
                confidence=0.95,
                strategy_used=strategy,
                cache_hit=True,
                tokens_consumed=630,  # L1 + L2
                cost_usd=self._calculate_cost(630, "haiku"),
                latency_ms=int((time.time() - start_time) * 1000),
                trace=trace
            )

        trace["layers"]["L2_cache"] = {"hit": False, "tokens": 310}

        # ==== LAYER 3: RETRIEVAL ====
        docs = await self.retriever.retrieve(query, strategy, game_id)

        trace["layers"]["L3_retrieval"] = {
            "strategy": strategy,
            "docs_retrieved": len(docs),
            "tokens": self._estimate_doc_tokens(docs, strategy)
        }

        # ==== LAYER 4: CRAG EVALUATION ====
        crag_result = await self.crag.evaluate(query, docs)

        # Apply CRAG filtering
        filtered_docs = crag_result.docs
        tokens_saved = crag_result.tokens_saved

        trace["layers"]["L4_crag"] = {
            "original_docs": len(docs),
            "filtered_docs": len(filtered_docs),
            "tokens_saved": tokens_saved,
            "source": crag_result.source
        }

        # Check if web augmentation is needed
        if crag_result.source in ["internal+web", "web"]:
            web_docs = await self._web_search_augment(query)
            filtered_docs.extend(web_docs)
            trace["layers"]["L4_crag"]["web_augmented"] = len(web_docs)

        # ==== LAYER 5: GENERATION ====
        if strategy == "PRECISE" and template == "resource_planning":
            # Multi-agent orchestration for PRECISE strategy queries
            ma_result = await self.multi_agent.execute(query, filtered_docs)
            answer = ma_result.final_answer
            confidence = ma_result.confidence
            gen_tokens = ma_result.tokens_used

            trace["layers"]["L5_generation"] = {
                "mode": "multi_agent",
                "agents": ["analyzer", "strategist", "validator"],
                "tokens": gen_tokens
            }
        else:
            # Standard generation with prompt builder
            prompt = self.generator.build(
                query=query,
                chunks=filtered_docs,
                template=template,
                strategy=strategy
            )

            llm_model = self._select_model(strategy)
            response = await self.llm_clients[llm_model].generate(
                prompt,
                max_tokens=self._get_max_tokens(template, strategy)
            )
            answer = response.content
            confidence = 0.0  # Set by validation
            gen_tokens = response.usage.total_tokens

            trace["layers"]["L5_generation"] = {
                "mode": "standard",
                "model": llm_model,
                "tokens": gen_tokens
            }

        # ==== LAYER 6: VALIDATION ====
        validation = await self.validator.validate(
            answer=answer,
            docs=filtered_docs,
            query=query,
            template=template,
            strategy=strategy
        )

        trace["layers"]["L6_validation"] = {
            "passed": validation.passed,
